                transactionList.appendChild(li);
            });

            miniStatementList.innerHTML = '';

            document.querySelectorAll('.notif-badge').forEach(badge => {
                badge.removeEventListener('click', badge._handler);
                const handler = function(e) {
                    e.stopPropagation();
                    const email = this.dataset.email;
                    if (emailNotifications.hasOwnProperty(email)) {
                        emailNotifications[email] = !emailNotifications[email];
                        this.textContent = emailNotifications[email] ? '🔔 on' : '🔕 off';
                    } else {
                        emailNotifications[email] = true;
                        this.textContent = '🔔 on';
                    }
                };
                badge._handler = handler;
                badge.addEventListener('click', handler);
            });
        }

        depositBtn.addEventListener('click', function() {
            const amount = parseFloat(depositAmount.value);
            const pin = depositPin.value;
            if (isNaN(amount) || amount <= 0) return alert('Enter valid amount');
            if (!validatePin(pin)) return alert('❌ Incorrect PIN');
            currentBalance += amount;
            addTransaction('Deposit', amount, 'credit');
            updateBalanceUI();
            alert(`✅ Deposited ₹${amount.toFixed(2)}`);
            depositAmount.value = ''; depositPin.value = '';
        });

        withdrawBtn.addEventListener('click', function() {
            const amount = parseFloat(withdrawAmount.value);
            const pin = withdrawPin.value;
            if (isNaN(amount) || amount <= 0) return alert('Enter valid amount');
            if (!validatePin(pin)) return alert('❌ Incorrect PIN');
            if (amount > currentBalance) return alert('❌ Insufficient balance');
            currentBalance -= amount;
            addTransaction('Withdrawal', amount, 'debit');
            updateBalanceUI();
            alert(`💸 Withdrawn ₹${amount.toFixed(2)}`);
            withdrawAmount.value = ''; withdrawPin.value = '';
        });

            if (isNaN(amount) || amount <= 0) return alert('Enter valid amount');
            if (!validatePin(pin)) return alert('❌ Incorrect PIN');
            if (amount > currentBalance) return alert('❌ Insufficient balance');
            currentBalance -= amount;
            addTransaction(`Transfer to ${account.substring(0,8)}...`, amount, 'debit');
